t

Light weight i18n support.

define strings in an object const i18n = { 'en': { hello: 'Hello {{name}}', }, 'zh_TW': { hello: '你好，{{name}}', }, props: ['name'], }

t('hello', { i18n, hello: 'world' }) // Hello World

export SAIHUBOT=zh_TW t('hello', { i18n, hello: 'world' })

// 你好，world
